Sub CreateCypressFrameworkHistorySlide()
    Dim pptApp As Object
    Set pptApp = CreateObject("PowerPoint.Application")
    pptApp.Visible = True

    ' Create a new presentation
    Dim pptPresentation As Object
    Set pptPresentation = pptApp.Presentations.Add

    ' Add a blank slide
    Dim pptSlide As Object
    Set pptSlide = pptPresentation.Slides.Add(1, ppLayoutText)

    ' Set the slide title
    pptSlide.Shapes(1).TextFrame.TextRange.Text = "Cypress Framework Development History"

    ' Define an array to hold your milestones
    Dim Milestones(1 To 5) As String
    Milestones(1) = "TDD Framework"
    Milestones(2) = "BDD Transition"
    Milestones(3) = "Enhancements & Reusability"
    Milestones(4) = "Parallel Feature Runs"
    Milestones(5) = "Complete Product Tests"

    ' Create a timeline or flowchart on the slide
    Dim LeftPos As Single
    LeftPos = 100

    For i = 1 To 5
        Set pptText = pptSlide.Shapes.AddTextEffect( _
            PresetTextEffect:=msoTextEffect1, Text:="▼", _
            FontName:="Arial Black", FontSize:=16, Left:=LeftPos, Top:=200)
        pptText.Rotation = 0
        pptText.TextFrame.TextRange.Text = Milestones(i)
        pptText.TextFrame.TextRange.Font.Bold = msoTrue
        pptText.TextFrame.TextRange.Font.Fill.ForeColor.RGB = RGB(0, 51, 102)
        LeftPos = LeftPos + 120
    Next i

    ' Save the presentation
    pptPresentation.SaveAs "YourFilePath\YourPresentationName.pptx"

    ' Clean up
    pptPresentation.Close
    Set pptPresentation = Nothing
    pptApp.Quit
    Set pptApp = Nothing
End Sub

# Use a Linux base image
FROM ubuntu:latest

# Update package list and install necessary dependencies
RUN apt-get update && \
    apt-get install -y curl wget gnupg && \
    apt-get clean

# Install Node.js 14 (or your desired version)
RUN curl -sL https://deb.nodesource.com/setup_14.x | bash -
RUN apt-get install -y nodejs

# Install Google Chrome
RUN wget -q -O - https://dl-ssl.google.com/linux/linux_signing_key.pub | apt-key add -
RUN sh -c 'echo "deb [arch=amd64] http://dl.google.com/linux/chrome/deb/ stable main" >> /etc/apt/sources.list.d/google-chrome.list'
RUN apt-get update && apt-get install -y google-chrome-stable

# Set Chrome flags to disable sandboxing for Docker
RUN sed -i 's|exec -a "$0" "$HERE/chrome" "$@"|exec -a "$0" "$HERE/chrome" "$@" --no-sandbox --disable-dev-shm-usage|g' /opt/google/chrome/google-chrome

# Set the default working directory
WORKDIR /app

# Install Cypress
RUN npm install cypress

# Copy your application files into the image
COPY . .

# Run Cypress tests (replace with your command)
CMD [ "npx", "cypress", "run" ]
