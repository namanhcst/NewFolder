import java.util.*;

public class Solution {
    
    // Precompute even powers of 2 up to some max
    private static List<Integer> evenPowersOfTwo(int max) {
        List<Integer> list = new ArrayList<>();
        int pow = 4; // smallest even power of 2
        while (pow <= max) {
            list.add(pow);
            pow *= 4; // next even power of 2
        }
        return list;
    }

    public static List<Integer> getMinimumKValues(List<Integer> arr) {
        List<Integer> result = new ArrayList<>();
        int maxVal = Collections.max(arr);
        List<Integer> powers = evenPowersOfTwo(maxVal);

        for (int num : arr) {
            int best = Integer.MAX_VALUE;
            boolean found = false;

            // try small = 1, 2, 3
            for (int small = 1; small <= 3; small++) {
                if (num < small) continue;
                int remaining = num - small;

                if (remaining == 0) {
                    best = Math.min(best, 1);
                    found = true;
                } else {
                    int count = 1; // for the small one
                    int rem = remaining;

                    // Greedily subtract largest even power of 2 until rem = 0
                    for (int i = powers.size() - 1; i >= 0; i--) {
                        int p = powers.get(i);
                        while (rem >= p) {
                            rem -= p;
                            count++;
                        }
                    }

                    if (rem == 0) {
                        best = Math.min(best, count);
                        found = true;
                    }
                }
            }

            result.add(found ? best : -1);
        }
        return result;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        List<Integer> arr = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            arr.add(sc.nextInt());
        }
        List<Integer> ans = getMinimumKValues(arr);
        for (int x : ans) {
            System.out.print(x + " ");
        }
    }
}
